14、横向滚动条事件
//必须用id ！！！！！！！！！！
let outDiv = document.getElementById('scrollPhoto');
        outDiv.onwheel = function(event){
          //禁止事件默认行为（此处禁止鼠标滚轮行为关联到"屏幕滚动条上下移动"行为）
          event.preventDefault();
          //设置鼠标滚轮滚动时屏幕滚动条的移动步长
          let step = 50;
          if(event.deltaY < 0){
            //向上滚动鼠标滚轮，屏幕滚动条左移
            this.scrollLeft -= step;
          } else {
            //向下滚动鼠标滚轮，屏幕滚动条右移
            this.scrollLeft += step;
          }
        }

16、修饰符

<!-- 即事件不是从内部元素触发的 -->
<div v-on:click.self="doThat">...</div>
